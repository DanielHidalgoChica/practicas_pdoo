def move(direction, valid_moves)
    size = valid_moves.size
    contained = valid_moves.include?(direction)
    if (size > 0 && !contained)
        valid_moves[0]
    else
        direction
    end
end

def receive_reward
    w_reward = Dice.weapons_reward
    s_reward = Dice.shields_reward
    w_reward.times do
        wnew = new_weapon
        receive_weapon(wnew)
    end
    s_reward.times do
        snew = new_shield
        receive_shield(snew)
    end
    extra_health =Dice.health_reward
    @health += extra_health
end

def to_s
    ret = "\nPlayer State" +
                "\nName: " + @name +
                "\nIntelligence: " + @intelligence.to_s +
                "\nStrength: " + @strength.to_s +
                "\nHealth: " + @health.to_s +
                "\nPosition: (" + @row.to_s + "," + @col.to_s + ")" +
                "\nConsecutive Hits: " + @consecutive_hits.to_s +
                "\nWeapons: "
    @weapons.each { |a_weapon| ret += "\n\t" + a_weapon.to_s }
    ret += "\nShields:"
    @shields.each { |a_shield| ret += "\n\t" + a_shield.to_s }
    ret
end
